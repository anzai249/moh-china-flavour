dalai.2 = {
	# New Dalai Lama
	type = country_event
	title = dalai.2.t
	desc = dalai.2.d
	flavor = dalai.2.f
	event_image = {
		video = "unspecific_signed_contract"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/ig_icons/devout.dds"
	duration = 3
	trigger = {
	}
	immediate = {
		remove_variable = dalai_lama_death
	}
	option = {
		name = dalai.2.a
		default_option = yes
		add_loyalists = {
			value = 0.2
			interest_group = ig:ig_devout
		}
		custom_tooltip = dalai.2.tt
		hidden_effect = {
			if = {
				limit = {
					global_var:dalai_lama_no = 10
				}
				create_character = {
					template = ecchi_tib_11_dalai_lama_character_template
					ruler = yes
					on_created = {
						set_variable = is_dalai_lama
					}
				}
				set_global_variable = {
					name = dalai_lama_no
					value = 11
				}
			}
			else_if = {
				limit = {
					global_var:dalai_lama_no = 11
				}
				create_character = {
					template = ecchi_tib_12_dalai_lama_character_template
					ruler = yes
					on_created = {
						set_variable = is_dalai_lama
					}
				}
				set_global_variable = {
					name = dalai_lama_no
					value = 12
				}
			}
			else_if = {
				limit = {
					global_var:dalai_lama_no = 12
				}
				create_character = {
					template = ecchi_tib_13_dalai_lama_character_template
					ruler = yes
					on_created = {
						set_variable = is_dalai_lama
					}
				}
				set_global_variable = {
					name = dalai_lama_no
					value = 13
				}
			}
			else_if = {
				limit = {
					global_var:dalai_lama_no = 13
				}
				create_character = {
					template = ecchi_tib_14_dalai_lama_character_template
					ruler = yes
					on_created = {
						set_variable = is_dalai_lama
					}
				}
				set_global_variable = {
					name = dalai_lama_no
					value = 14
				}
			}
			else_if = {
				limit = {
					OR = {
						global_var:dalai_lama_no > 13
						global_var:dalai_lama_no < 10
					}
				}
				create_character = {
					last_name = "Gyatso"
					age = 5
					ruler = yes
					interest_group = ig_devout
					ideology = ideology_theocrat
					on_created = {
						set_variable = is_dalai_lama
					}
				}
				set_global_variable = {
					name = dalai_lama_no
					value = 15
				}
			}
		}
	}
}
