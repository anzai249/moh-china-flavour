china_communism.2 = {
	type = country_event
	placement = scope:canton_scope
	event_image = {
		video = "unspecific_steam_ship"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	title = china_communism.2.t
	desc = china_communism.2.d
	flavor = china_communism.2.f

	duration = 3

	dlc = dlc0moh

	trigger = {
		c:CHI ?= this
		has_technology_researched = socialism
		any_scope_character = {
			has_ideology = ideology:ideology_moh_kmt
		}
		NOT = {
			any_scope_character = {
				has_template = chi_moh_sun_yat_sen_character_template
			}
		}
		OR = {
			any_scope_character = {
				has_ideology = ideology:ideology_communist
			}
			any_scope_character = {
				has_ideology = ideology:ideology_vanguardist
			}
		}
		NOT = {
			has_variable = canton_coup_happened
		}
		always = no
	}

	immediate = {
		set_variable = canton_coup_happened
		s:STATE_GUANGDONG = {
			random_scope_state = {
				save_scope_as = canton_scope
			}
		}
	}

	option = {
		name = china_communism.2.a
		default_option = yes
		custom_tooltip = {
			text = china_communism.2.a_tt
			set_variable = canton_coup_anti_communist
		}
		divide_kmt = yes
		c:CHI = {
			add_loyalists = {
				value = -0.2
				pop_type = laborers
			}
		}
	}

	option = {
		name = china_communism.2.b
		custom_tooltip = {
			text = china_communism.2.b_tt
			set_variable = canton_coup_support_communist
		}
		c:CHI = {
			add_radicals = {
				value = -0.2
				pop_type = laborers
			}
		}
	}
}
