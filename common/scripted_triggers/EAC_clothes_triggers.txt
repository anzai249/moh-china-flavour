japanese_imperial_clothes_trigger = {
	exists = scope:character
	scope:character = {
		is_ruler = yes
		exists = home_country
     	home_country = {
     		has_law = law_type:law_monarchy
			has_variable = meiji_var
     	}
		OR = {
			AND = {
				culture = {
					has_discrimination_trait = yamato
				}
				OR = {
					NOT = {
						exists = c:JAP
					}
					AND = {
						exists = c:JAP
						NOT = { owner = c:JAP }
					}
				}
			}
			AND = {
				exists = c:JAP
				home_country = {
					this = c:JAP
					NOT = { has_variable = meiji_restoration_complete }
				}
			}
		}
	}
}

			AND = {

					ruler = {
						has_culture = cu:manchu
					}
					has_law = law_type:law_monarchy
					NOT = { 
						has_law = law_type:law_multicultural
					}
				}
				culture = {
					OR = {
						has_discrimination_trait = east_asian_heritage
						has_discrimination_trait = north_asian_heritage
					}
					NOT = {
						this = cu:japanese
						this = cu:korean
						this = cu:vietnamese
					}
				}
				NOT = {
					AND = {
						is_pop_type = clergymen
						religion = {
							has_discrimination_trait = buddhist
							NOT = {
								this = rel:mahayana
							}
						}
					}
				}
			}
			AND = {
				NOT = {
					owner = { 
						ruler = { 
							culture = { 
								has_discrimination_trait = han_chinese 
							}
						}
						cu:manchu = {
							NOT = { 
								culture_is_discriminated_in = scope:pop.owner 
							}
						}
					}
				}
				culture = cu:manchu
			}
			AND = {
				exists = c:CHI
				c:CHI = {
					ruler = {
						has_culture = cu:manchu
					}
					has_law = law_type:law_monarchy
					NOT = { 
						has_law = law_type:law_multicultural
					}
				}
				owner = {
					ruler = {
						culture = {
							has_discrimination_trait = east_asian_heritage
						}
					}
					has_law = law_type:law_monarchy
					has_law = law_type:law_autocracy
					NOT = {
						has_law = law_type:law_multicultural
					}
				}
				culture = {
					OR = {
						has_discrimination_trait = east_asian_heritage
						has_discrimination_trait = north_asian_heritage
					}
					NOT = {
						this = cu:japanese
						this = cu:korean
						this = cu:vietnamese
						this = cu:ainu
					}
				}
			}
		}
	}
