print_reformer_newspaper_button = {
	name = "print_reformer_newspaper"
	desc = "print_reformer_newspaper_desc"
	visible = {
		has_journal_entry = je_wuxu
	}
	possible = {
		custom_tooltip = {
			text = print_reformer_newspaper_button.tt.1
			var:wuxu_newspaper_time >= 20
		}
	}
	effect = {
		custom_tooltip = {
			text = print_reformer_newspaper_button.tt.2
			set_variable = {
				name = wuxu_newspaper_time
				value = 0
			}
		}
		add_modifier = {
			name = newspaper_campaign_modifier
			months = 3
			multiplier = money_amount_multiplier_small
		}
	}
	ai_chance = {
		base = 15
		modifier = {
			trigger = {
				has_strategy = ai_strategy_conservative_agenda
			}
			add = -10
		}
		modifier = {
			trigger = {
				has_strategy = ai_strategy_maintain_mandate_of_heaven
			}
			add = 10
		}
	}
}

china_ming_button = {
	name = "china_ming"
	desc = "china_ming_desc"
	visible = {
		has_journal_entry = je_new_dynasty
	}
	possible = {
		NOT = {
			has_variable = china_ming
		}
		country_has_primary_culture = cu:han
		ruler = {
			has_culture = cu:han
		}
	}
	effect = {
		set_variable = china_ming
		show_as_tooltip = {
			create_character = {
				
			}
		}
		hidden_effect = {
			heir = {
				kill_character = yes
			}
			if = {
				limit = {
					year < 1869
				}
				create_character = {
					last_name = Zhu
					first_name = Shugui
					culture = cu:han
					ruler = yes
					historical = yes
				}
			}
			else_if = {
				limit = {
					year < 1891
				}
				create_character = {
					last_name = Zhu
					first_name = Chengduan
					culture = cu:han
					ruler = yes
					historical = yes
				}
			}
			else = {
				create_character = {
					last_name = Zhu
					first_name = Yuxun
					culture = cu:han
					ruler = yes
					historical = yes
				}
			}
			
		}
	}
	ai_chance = {
		base = 1
	}
}
