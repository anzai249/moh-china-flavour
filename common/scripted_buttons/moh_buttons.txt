china_ming_button = {
    name = "china_ming"
    desc = "china_ming_desc"
    visible = {
        has_journal_entry = je_new_dynasty
    }
    possible = {
        NOT = {
            has_variable = china_ming
        }
        country_has_primary_culture = cu:han
        ruler = {
            has_culture = cu:han
        }
    }
    effect = {
        set_variable = china_ming
        show_as_tooltip = {
            create_character = {

            }
        }
        if = {
            limit = {
                country_has_primary_culture = cu:manchu
            }
            remove_primary_culture = cu:manchu
        }
        remove_8_banners = yes
        hidden_effect = {
            heir = {
                kill_character = yes
            }
            if = {
                limit = {
                    year < 1869
                }
                create_character = {
                    last_name = Zhu
                    first_name = Shugui
                    culture = cu:han
                    ruler = yes
                    historical = yes
                }
            }
            else_if = {
                limit = {
                    year < 1891
                }
                create_character = {
                    last_name = Zhu
                    first_name = Chengduan
                    culture = cu:han
                    ruler = yes
                    historical = yes
                }
            }
            else = {
                create_character = {
                    last_name = Zhu
                    first_name = Yuxun
                    culture = cu:han
                    ruler = yes
                    historical = yes
                }
            }

        }
    }
    ai_chance = {
        value = 1
    }
}

# Gaitu Guiliu
gaitu_guiliu_button_1 = {
    name = "gaitu_guiliu_button_1"
    desc = "gaitu_guiliu_button_1_desc"

    visible = { always = yes }

    possible = {
        NOT = {
            custom_tooltip = {
                text = gaitu_guiliu_cooldown_tt
                has_variable = gaitu_guiliu_cooldown_var
            }
        }
        any_scope_state = {
            has_state_trait = state_trait_chieftains
            any_scope_building = {
                is_building_type = building_urban_center
                level >= 10
            }
            any_scope_building = {
                is_building_type = building_government_administration
                level >= 15
            }
            NOT = {
                has_modifier = gaitu_guiliu_cooldown_modifier
            }
        }
    }

    ai_chance = {
        value = 20
        if = {
            limit = {
                OR = {
                    is_at_war = yes
                    has_journal_entry = je_incorporate_the_west
                }
            }
            add = -50
        }
    }

    effect = {
        set_variable = { name = gaitu_guiliu_cooldown_var years = 2 }
        random_scope_state = {
            limit = {
                has_state_trait = state_trait_chieftains
                any_scope_building = {
                    is_building_type = building_urban_center
                    level >= 10
                }
                any_scope_building = {
                    is_building_type = building_government_administration
                    level >= 15
                }
                NOT = {
                    has_modifier = gaitu_guiliu_cooldown_modifier
                }
            }
            add_modifier = {
                name = gaitu_guiliu_cooldown_modifier
            }
        }
        show_as_tooltip = {
            random_scope_state = {
                limit = {
                    has_state_trait = state_trait_chieftains
                    OR = {
                        any_scope_building = {
                            is_building_type = building_urban_center
                            level < 10
                        }
                        any_scope_building = {
                            is_building_type = building_government_administration
                            level < 15
                        }
                    }
                }
                add_modifier = {
                    name = gaitu_guiliu_cooldown_modifier
                }
            }
        }
    }
}

gaitu_guiliu_button_2 = {
    name = "gaitu_guiliu_button_2"
    desc = "gaitu_guiliu_button_2_desc"

    visible = { always = yes }

    possible = {
        NOT = {
            custom_tooltip = {
                text = gaitu_guiliu_cooldown_tt
                has_variable = gaitu_guiliu_cooldown_var
            }
        }
        any_scope_state = {
            has_state_trait = state_trait_chieftains
            has_modifier = gaitu_guiliu_cooldown_modifier
            OR = {
                any_scope_building = {
                    is_building_type = building_urban_center
                    level >= 15
                }
                any_scope_building = {
                    is_building_type = building_railway
                    level >= 1
                }
            }
            any_scope_building = {
                is_building_type = building_government_administration
                level >= 20
            }
        }
    }

    ai_chance = {
        value = 20
        if = {
            limit = {
                OR = {
                    is_at_war = yes
                    has_journal_entry = je_incorporate_the_west
                }
            }
            add = -50
        }
    }

    effect = {
        set_variable = { name = gaitu_guiliu_cooldown_var years = 2 }
        random_scope_state = {
            limit = {
                has_state_trait = state_trait_chieftains
                has_modifier = gaitu_guiliu_cooldown_modifier
                OR = {
                    any_scope_building = {
                        is_building_type = building_urban_center
                        level >= 15
                    }
                    any_scope_building = {
                        is_building_type = building_railway
                        level >= 1
                    }
                }
                any_scope_building = {
                    is_building_type = building_government_administration
                    level >= 20
                }
            }
            state_region = {
                remove_state_trait = state_trait_chieftains
            }
            remove_modifier = gaitu_guiliu_cooldown_modifier
            if = {
                limit = {
                    this.state_region = s:STATE_SICHUAN
                }
                state_region = {
                    add_state_trait = state_trait_the_red_basin
                }
            }
        }
        show_as_tooltip = {
            random_scope_state = {
                limit = {
                    has_state_trait = state_trait_chieftains
                    AND = {
                        any_scope_building = {
                            is_building_type = building_urban_center
                            level < 15
                        }
                        any_scope_building = {
                            is_building_type = building_railway
                            level < 1
                        }
                    }
                    any_scope_building = {
                        is_building_type = building_government_administration
                        level < 20
                    }
                }
                state_region = {
                    remove_state_trait = state_trait_chieftains
                }
                remove_modifier = gaitu_guiliu_cooldown_modifier
                if = {
                    limit = {
                        this.state_region = s:STATE_SICHUAN
                    }
                    state_region = {
                        add_state_trait = state_trait_the_red_basin
                    }
                }
            }
        }
    }
}
