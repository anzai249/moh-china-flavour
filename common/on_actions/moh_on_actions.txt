moh_linzexu = {
	effect = {
		if = {
			limit = {
				year >= 1842
				any_scope_character = {
					has_variable = is_linzexu
					is_immortal = yes
				}
			}
			random_scope_character = {
				limit = {
					has_variable = is_linzexu
				}
				set_character_immortal = no
			}
		}
	}
}

on_half_yearly_pulse_country = {

	}
}

moh_sunyatsen_je = {
	effect = {
		if = {
			limit = {
				has_global_variable = china_shatters
				country_rank = 1
			}
			if = {
				limit = {
					any_country = {
						sun_yatsen_trigger_30 = yes
					}
				}
				random_country = {
					limit = {
						sun_yatsen_trigger_30 = yes
					}
					sun_yatsen_goto = yes
				}
			}
			else_if = {
				limit = {
					any_country = {
						sun_yatsen_trigger_25 = yes
					}
				}
				random_country = {
					limit = {
						sun_yatsen_trigger_25 = yes
					}
					sun_yatsen_goto = yes
				}
			}
			else_if = {
				limit = {
					any_country = {
						sun_yatsen_trigger_20 = yes
					}
				}
				random_country = {
					limit = {
						sun_yatsen_trigger_20 = yes
					}
					sun_yatsen_goto = yes
				}
			}
			else_if = {
				limit = {
					any_country = {
						sun_yatsen_trigger_15 = yes
					}
				}
				random_country = {
					limit = {
						sun_yatsen_trigger_15 = yes
					}
					sun_yatsen_goto = yes
				}
			}
			else_if = {
				limit = {
					any_country = {
						sun_yatsen_trigger_10 = yes
					}
				}
				random_country = {
					limit = {
						sun_yatsen_trigger_10 = yes
					}
					sun_yatsen_goto = yes
				}
			}
			else = {
				random_country = {
					limit = {
						country_rank = rank_value:great_power
						NOT = {
							has_law = law_type:law_monarchy
						}
					}
					sun_yatsen_goto = yes
				}
			}
		}
	}
}

moh_qing_monarchy_yearly_events = {
	effect = {
		if = {
			limit = {
				owner = {
					ruler = {
						OR = {
							has_template = chi_tongzhi_template
							has_template = chi_guangxu_template
						}
					}
				}
			}
            kill_character = {
                value  = yes
                hidden = yes
            }
		}
	}
}

				value = 1
			}
		}
		if = {
			limit = {
				has_template = ecchi_chi_wang_jingwei_character_template
			}
			set_global_variable = {
				name = wangjingwei_dead
				value = 1
			}
		}
	}
}

on_revolution_start = {
	on_actions = {
		moh_on_revolution_start
	}
}

moh_on_revolution_start = {
	effect = {
		if = {
			limit = {
				c:CHI ?= ROOT
				scope:target = {
					has_law = law_type:law_monarchy
					NOT = {
						has_journal_entry = je_new_dynasty
						has_variable = china_ming
						has_variable = china_shun
						has_variable = china_later_jin
						has_variable = china_han
						has_variable = china_qing
					}
				}
			}
			scope:target = {
				trigger_event = warlord_china_events.500
			}
		}
	}
}

on_wargoal_enforced = {
	on_actions = {
		moh_self_strengthing_war
	}
}

moh_self_strengthing_war = {
	effect = {
		if = {
			limit = {
				scope:target = {
					has_journal_entry = je_self_strengthing
				}
			}
			scope:target = {
				if = {
					limit = {
						has_variable = self_strengthening_lost_wars
					}
					change_variable = {
						name = self_strengthening_lost_wars
						value = 1
					}
				}
			}
		}
	}
}

on_character_creation = {
	on_actions = {
		moh_self_strengthening_character_creation
	}
}

moh_self_strengthening_character_creation = {
	effect = {
		if = {
			limit = {
				owner = {
					has_journal_entry = je_self_strengthing
				}
				is_historical = no
			}
			random_list = {
				75 = 0
				25 = {
					set_variable = self_strengthening_character
				}
			}
		}
	}
}
