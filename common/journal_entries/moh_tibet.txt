je_moh_tibet_look_for_new_dalai_lama = {
	icon = "gfx/interface/icons/ig_icons/devout.dds"
	group = je_group_historical_content
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					any_state = {
						owner = ROOT
						turmoil >= 0.1
					}
				}
				change_variable = {
					name = look_for_new_dalai_lama_progress
					add = 1
				}
			}
			else = {
				change_variable = {
					name = look_for_new_dalai_lama_progress
					add = 2
				}
			}
		}
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	fail = {
		NOT = {
			has_law = law_type:law_theocracy
		}
	}
	on_complete = {
		trigger_event = {
			id = dalai.2
		}
	}
	on_fail = {
	}
	current_value = {
		value = root.var:look_for_new_dalai_lama_progress
	}
	goal_add_value = {
		add = 60
	}
	weight = 100
	progressbar = yes
	should_be_pinned_by_default = yes
}
