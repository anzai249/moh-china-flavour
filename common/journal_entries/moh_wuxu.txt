je_wuxu = {
	icon = "gfx/interface/icons/law_icons/landed_voting.dds"
	group = je_group_historical_content
	scripted_progress_bar = wuxu_progress_bar
	scripted_button = print_reformer_newspaper_button
	scripted_button = steady_advancement_button
	scripted_button = dismissed_officials_petition_button
	immediate = {
		set_variable = {
			name = wuxu_reformers
			value = 120
		}
		set_variable = {
			name = wuxu_conservatives
			value = 100
		}
		set_variable = {
			name = wuxu_balance_passive_var
			value = 0
		}
		set_variable = {
			name = wuxu_newspaper_time
			value = 5
		}
	}
	modifiers_while_active = {
		modifier_chi_wuxu
	}
	complete = {
		custom_tooltip = je_wuxu.tt.1
		OR = {
			has_law = law_type:law_landed_voting
			has_law = law_type:law_wealth_voting
			has_law = law_type:law_universal_suffrage
			has_law = law_type:law_census_voting
		}
		has_law = law_type:law_professional_army
		NOT = {

			has_law = law_type:law_isolationism
		}
		custom_tooltip = {
			text = je_wuxu.tt.4
			any_scope_building = {
				filter = {
					is_building_type = building_barracks
				}
				percent > 0.75
				has_active_production_method = pm_advanced_tactics_training
			}
		}
	}
	fail = {
		custom_tooltip = {
			text = je_wuxu.tt.3
			var:wuxu_reformers <= 0
		}
	}
	on_complete = {
		custom_tooltip = je_wuxu.tt.5
		trigger_event = {
			id = wuxu.2
		}
	}
	on_fail = {
		custom_tooltip = je_wuxu.tt.2
		trigger_event = {
			id = wuxu.1
		}
	}
	on_weekly_pulse = {
		effect = {
			change_variable = {
				name = wuxu_newspaper_time
				add = 1
			}
		}
	}
	current_value = {
		value = var:wuxu_reformers
	}
	weight = 100
	should_be_pinned_by_default = yes
}
