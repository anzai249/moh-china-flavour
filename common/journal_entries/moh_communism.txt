je_opium_communism = {
	icon = "gfx/interface/icons/goods_icons/opium.dds"
	group = je_group_historical_content
	scripted_button = ban_opium_button
	scripted_button = unban_opium_button
	should_be_pinned_by_default = yes
	progressbar = yes
	timeout = 3650
	weight = 100
	is_shown_when_inactive = {
		c:CHI ?= this
		has_law = law_type:law_council_republic
		any_primary_culture = {
			has_cultural_obsession = opium
		}
	}
	possible = {
		c:CHI ?= this
		has_law = law_type:law_council_republic
		any_primary_culture = {
			has_cultural_obsession = opium
		}
		is_revolutionary = no
	}
	immediate = {
		set_variable = {
			name = opium_communism_progress
			value = 0
		}
	}
	complete = {
		custom_tooltip = {
			text = je_opium_communism_complete_tt
			var:opium_communism_progress >= 100
		}
	}
	on_complete = {
		every_primary_culture = {
			remove_cultural_obsession = opium
		}
		add_modifier = {
			name = modifier_ban_opium_communism
			years = 10
		}
		remove_banned_goods = g:opium
	}
	goal_add_value = {
		add = 100
	}
	current_value = {
		value = var:opium_communism_progress
	}
	on_monthly_pulse = {
		random_events = {
			100 = 0
			5 = moh_historical.202
			5 = moh_historical.201
		}
		effect = {
			if = {
				limit = {
					is_banning_goods = opium
				}
				change_variable = {
					name = opium_communism_progress
					add = 1
				}
			}
		}
	}
}

je_anti_encirclement = {
	icon = "gfx/interface/icons/invention_icons/socialism.dds"
	group = je_group_global_international_situations
	weight = 100
	should_be_pinned_by_default = yes
	immediate = {
		set_global_variable = {
			name = moh_china_civil_war_score_kmt
			value = 0
		}
		set_global_variable = {
			name = moh_china_civil_war_score_ccp
			value = 0
		}
		set_global_variable = {
			name = moh_china_civil_war_cooldown_months
			value = 12
		}
	}
    is_shown_in_lobby = {
		has_global_variable = ccp_rise_up_happened
        OR = {
        	c:CHI ?= this
        	c:CCP ?= this
        }
    }
	should_be_involved = {
		country_has_primary_culture = cu:han
	}
	should_show_when_not_involved = {
		always = no
	}
	modifiers_while_active = {
		modifier_moh_communist_china_guerilla_warfare
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_anti_encirclement_status_countdown
				trigger = {
					global_var:moh_china_civil_war_cooldown_months > -1
				}
			}
			triggered_desc = {
				desc = je_anti_encirclement_status
				trigger = {
					global_var:moh_china_civil_war_cooldown_months <= -1
				}
			}
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					global_var:moh_china_civil_war_cooldown_months = 0
				}
				change_global_variable = {
					name = moh_china_civil_war_cooldown_months
					add = -1
				}
				c:CCP = {
					create_diplomatic_play = {
						target_state = c:CHI.capital
						war = no
						annex_as_civil_war = yes
						type = dp_revolution
					}
					activate_law = law_type:law_single_party_state
				}
			}
			else = {
				change_global_variable = {
					name = moh_china_civil_war_cooldown_months
					add = -1
				}
			}
		}
	}
	complete = {
		custom_tooltip = {
			text = je_anti_encirclement_complete_tt
			OR = {
				global_var:moh_china_civil_war_score_kmt >= 1
				global_var:moh_china_civil_war_score_ccp >= 5
				NAND = {
					exists = c:CHI
					exists = c:CCP
				}
			}
		}
	}
	on_complete = {
		random_diplomatic_play = {
			limit = {
				target_is = c:CHI
				initiator_is = c:CCP
			}
			end_play = yes
		}
		c:CHI = {
			create_bidirectional_truce = {
				country = c:CCP
				months = 24
			}
		}
		c:CCP = {
			if = {
				limit = {
					has_modifier = modifier_communist_china_soviet_base_area
				}
				remove_modifier = modifier_communist_china_soviet_base_area
			}
		}
		if = {
			limit = {
				global_var:moh_china_civil_war_score_kmt >= 1
			}
			s:STATE_XIAN = {
				set_owner_of_provinces = {
					country = c:CCP
					provinces = { x1FCA38 x08880B x8A81C7 x6E8E64 xFE1D86 xA7D4E7
					x32EDB3 xF40585 x434092 x5E3640 x3CE218 x826987 xCBACD2 x38D512 
					x09BD45 x3A0F5D xADCF92 xD01080 x7ED6E0 x2B47B0 }
				}
			}
			transfer_state = {
				STATE = s:STATE_JIANGXI
				GIVER = c:CCP
				TAKER = c:CHI
			}
			c:CHI = {
				trigger_event = {
					id = china_communism.3
				}
			}
			c:CCP = {
				trigger_event = {
					id = china_communism.4
				}
				create_military_formation = {
					type = army
					hq_region = sr:region_south_china
					name = ig_red_army
					combat_unit = {
						type = unit_type:combat_unit_type_line_infantry
						state_region = s:STATE_JIANGXI
						count = 15
					}
					combat_unit = {
						type = unit_type:combat_unit_type_hussars
						state_region = s:STATE_NINGXIA
						count = 5
					}
					combat_unit = {
						type = unit_type:combat_unit_type_cannon_artillery
						state_region = s:STATE_NINGXIA
						count = 5
					}
				}
			}
		}
		else_if = {
			limit = {
				global_var:moh_china_civil_war_score_ccp >= 5
			}
			transfer_state = {
				STATE = s:STATE_JIANGXI
				GIVER = c:CHI
				TAKER = c:CCP
			}
			transfer_state = {
				STATE = s:STATE_FUJIAN
				GIVER = c:CHI
				TAKER = c:CCP
			}
			transfer_state = {
				STATE = s:STATE_HUNAN
				GIVER = c:CHI
				TAKER = c:CCP
			}
			transfer_state = {
				STATE = s:STATE_SOUTHERN_ANHUI
				GIVER = c:CHI
				TAKER = c:CCP
			}
			transfer_state = {
				STATE = s:STATE_EASTERN_HUBEI
				GIVER = c:CHI
				TAKER = c:CCP
			}
			transfer_state = {
				STATE = s:STATE_ZHEJIANG
				GIVER = c:CHI
				TAKER = c:CCP
			}
			transfer_state = {
				STATE = s:STATE_HAINAN
				GIVER = c:CHI
				TAKER = c:CCP
			}
			c:CCP = {
				every_scope_state = {
					limit = {
						is_homeland = han
					}
					set_state_type = incorporated
				}
			}
		}
	}
}
