je_father_of_the_republic = {
	icon = "gfx/interface/icons/event_icons/white_sun.dds"
	group = je_group_global_international_situations
	widget = {
		gui = "gui/journal_entry_widgets/moh_china_widgets.gui"
		name = "widget_je_father_of_the_republic"
		container = "custom_widget_container_je_icon"
	}
	is_shown_when_inactive = {
		always = yes
	}
    is_shown_in_lobby = {
        c:CHI ?= THIS
    }
	should_be_involved = {
		country_has_primary_culture = cu:han
		country_is_in_east_asia = yes
		# has_journal_entry = je_reunify_china
	}
	should_show_when_not_involved = {
		country_has_primary_culture = cu:han
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = father_of_the_republic_status_1
				trigger = {
					any_character = {
						has_variable = sun_yatsen_character
						owner = {
							NOT = {
								country_has_primary_culture = cu:han
							}
						}
					}
				}
			}
			# triggered_desc = {
			# 	desc = father_of_the_republic_status_2
			# 	trigger = {
			# 		global_var:sun_yatsen_in = {
			# 			has_journal_entry = je_reunify_china
			# 		}
			# 	}
			# }
		}
	}
	immediate = {
		random_country = {
			limit = {
				has_state_in_state_region = STATE_HAWAIIAN_ISLANDS
			}
			create_character = {
				template = chi_moh_sun_yat_sen_character_template
			}
			set_global_variable = {
				name = sun_yatsen_in
				value = this
			}
		}
		if = {
			limit = {
				NOT = {
					has_global_variable = sun_yatsen
				}
				c:BEI ?= this
			}
			je:je_father_of_the_republic ?= {
				clear_variable_list = com_journal_characters
				add_to_variable_list = {
					name = com_journal_characters
					target = global_var:sun_yatsen
				}
			}
		}
	}
	possible = {
		any_country = {
			has_journal_entry = je_reunify_china
		}
		# year > 1886
		NOT = {
			has_global_variable = sun_yatsen_dead
		}
	}
	complete = {
		custom_tooltip = {
			text = je_father_of_the_republic.tt.complete
			exists = c:CHI
		}
	}
	on_complete = {
		custom_tooltip = {
			text = je_father_of_the_republic.tt.on.complete
			random_character = {
				limit = {
					has_variable = sun_yatsen_character
				}
				transfer_character = c:CHI
				set_as_interest_group_leader = yes
				add_character_role = character_role_ig_leader
				set_character_as_ruler = yes
				set_character_immortal = no
			}
		}
		
		every_country = {
			limit = {
				has_modifier = modifier_father
			}
			remove_modifier = modifier_father
		}
	}
	fail = {
		custom_tooltip = {
			text = je_father_of_the_republic.tt.fail
			has_global_variable = sun_yatsen_dead
		}
	}
	on_fail = {
		every_country = {
			limit = {
				has_modifier = modifier_father
			}
			remove_modifier = modifier_father
		}
	}
	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = {
						any_character = {
							has_variable = sun_yatsen_character
						}
					}
					NOT = {
						has_global_variable = sun_yatsen_dead
					}
				}
				random_country = {
					limit = {
						has_state_in_state_region = STATE_HAWAIIAN_ISLANDS
					}
					create_character = {
						template = chi_moh_sun_yat_sen_character_template
					}
					set_global_variable = {
						name = sun_yatsen_in
						value = this
					}
				}
			}
			random_country = {
				limit = {
					NOT = {
						has_modifier = modifier_father
					}
					has_journal_entry = je_reunify_china
					any_scope_character = {
						has_variable = sun_yatsen_character
					}
				}
				add_modifier = modifier_father
			}
			random_country = {
				limit = {
					has_modifier = modifier_father
					has_journal_entry = je_reunify_china
					NOT = {
						any_scope_character = {
							has_variable = sun_yatsen_character
						}
					}
				}
				remove_modifier = modifier_father
			}
		}
	}
	on_monthly_pulse = {
		random_events = {
			10 = 0
			5 = moh_historical.1
		}
	}
	weight = 500
	should_be_pinned_by_default_involved = yes
	should_be_pinned_by_default_uninvolved_or_context = yes
}
