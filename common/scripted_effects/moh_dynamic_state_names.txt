# All dynamic names should be applied on a strategic region level for ease of use and to avoid overlapping script
INJECT:assign_state_hub_dynamic_names = {
	### Mandate of Heaven
	if = {
        limit = {
        	region = sr:region_south_china
        }
		assign_state_name_region_south_china = yes
	}
}

assign_state_name_region_south_china = {
    if = {
        limit = {
            state_region = {
                s:STATE_SHAOZHOU ?= this
            }
        }
        STATE_SHAOZHOU_state_name_assign = yes
    }
}

STATE_SHAOZHOU_state_name_assign = {
    if = {
        limit = {
            owner = {
                NOT = {
                    country_has_primary_culture = cu:han
                }
            }
        }
		reset_state_name = yes
        reset_hub_names = yes
    }
}
